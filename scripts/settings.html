You can change the look and behavior of SumatraPDF by editing SumatraPDF.dat settings file.

The file is stored in %APPDATA%\SumatraPDF directory for the installed version or in the same directory as SumataPDF.exe executable for the portable version.

The file is in a simple text format. Here's what different settings mean and what is their default value:

All values in this structure can't be modified from within the UI and are for 
experimental or expert settings. They're overwritten by the content of 
SumatraPDF-user.ini if that file exists.
UserPrefs [
  background color of the non-document windows, traditionally yellow
  MainWindowBackground = #fff200

  whether the Esc key will exit SumatraPDF same as 'q'
  EscToExit = false

  whether opening a new document should happen in an already running 
  SumatraPDF instance so that there's only one process and documents 
  aren't opend twice
  ReuseInstance = false

  these values allow to customize the UI used for fixed page documents 
  (PDF, XPS, DjVu, etc.)
  FixedPageUI [
    color value with which black (text) will be substituted
    TextColor = #000000

    color value with which white (background) will be 
    substituted
    BackgroundColor = #ffffff

    colors to use for the gradient from top to bottom (stops 
    will be inserted at regular intervals throughout the 
    document); currently only up to three colors are 
    supported; the idea behind this experimental feature is 
    that the background might allow to subconsciously 
    determine reading progress; suggested values: #2828aa 
    #28aa28 #aa2828
    GradientColors =

    horizontal and vertical distance between two pages in 
    facing and book view modes
    PageSpacing = 0 0
  ]

  these values allow to customize the UI used for ebooks (with 
  UseFixedPageUI disabled)
  EbookUI [
    whether the UI used for PDF documents will be used for 
    ebooks as well (enables printing and searching, disables 
    automatic reflow)
    UseFixedPageUI = false

    color for text
    TextColor = #5f4b32

    color of the background (page)
    BackgroundColor = #d9f0fb
  ]

  these values allow to customize the UI used for images and comic 
  books
  ImageOnlyUI [
    horizontal and vertical distance between two pages in 
    facing and book view modes
    PageSpacing = 0 0
  ]

  these values allow to customize the UI used for CHM documents (with 
  UseFixedPageUI disabled)
  ChmUI [
    whether the UI used for PDF documents will be used for 
    CHM documents as well
    UseFixedPageUI = false
  ]

  this list contains a list of additional external viewers for various 
  file types (multiple entries of the same format are recognised)
  ExternalViewers [
    command line with which to call the external viewer, may 
    contain %p for page numer and %1 for the file name
    CommandLine =

    name of the external viewer to be shown in the menu 
    (implied by CommandLine if missing)
    Name =

    filter for which file types the menu item is to be shown 
    (e.g. "*.pdf;*.xps"; "*" if missing)
    Filter =
  ]

  zoom levels which zooming steps through, excluding the virtual zoom 
  levels fit page, fit content and fit width (minimal allowed value is 
  8.33 and maximum allowed value is 6400)
  ZoomLevels = 8.33 12.5 18 25 33.33 50 66.67 75 100 125 150 200 300 400 600 800 1000 1200 1600 2000 2400 3200 4800 6400

  zoom step size in percents relative to the current zoom level (if 
  zero or negative, the values from ZoomLevels are used instead)
  ZoomIncrement = 0

  these values allow to override the default settings in the Print 
  dialog
  PrinterDefaults [
    default value for scaling (shrink, fit, none)
    PrintScale = shrink

    default value for the compatibility option
    PrintAsImage = false
  ]

  sizes of the top, right, bottom and left margin (in that order) 
  between window and document
  WindowMargin = 2 4 2 4

  these values allow to customize how the forward search highlight 
  appears
  ForwardSearch [
    when set to a positive value, the forward search 
    highlight style will be changed to a rectangle at the 
    left of the page (with the indicated amount of margin 
    from the page margin)
    HighlightOffset = 0

    the width of the highlight rectangle for when 
    HighlightOffset is set
    HighlightWidth = 15

    the color used for the forward search highlight
    HighlightColor = #ff8165

    whether the forward search highlight will remain visible 
    until the next mouse click instead of fading away 
    instantly
    HighlightPermanent = false
  ]
]
